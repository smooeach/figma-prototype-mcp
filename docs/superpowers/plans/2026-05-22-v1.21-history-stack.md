# v1.21 — In-memory history stack + `proto_get_last_history` Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an in-memory FIFO ring buffer (cap 10) that records every successful `proto_*` tool call, expose a new `proto_get_last_history` MCP tool to query it, and prepend the Korean usage-restriction disclaimer to all 4 proto_* tool descriptions.

**Architecture:** A new pure module `src/server/history.ts` (HistoryStore class + summarizeResult helper + types) with no plugin coupling. proto_* handlers in `src/server/tools.ts` gain a single `historyStore.record(...)` call after `session.sendCommand` resolves. `proto_get_last_history` is registered alongside the existing 9 tools. `registerToolHandlers` signature extends from `(mcp, session)` to `(mcp, session, historyStore)`; the store is instantiated once in `src/server/index.ts` and injected.

**Tech Stack:** Node's `crypto.randomUUID()` for ids, `Date.now()` for timestamps, zod for the new input schema, Vitest for tests. No new npm dependencies.

**Source spec:** `docs/superpowers/specs/2026-05-22-v1.21-history-stack-design.md`

**Designer milestone protocol:** Each task ends with a one-line "디자이너 요약" note for the UXUI designer reader. Implementation subagents must include this note in their report.

```ts
import { randomUUID } from "node:crypto";

export type ProtoToolName = "proto_wire" | "proto_overlay" | "proto_scroll";

export interface HistoryEntry {
  historyId: string;
  timestamp: number;
  tool: ProtoToolName;
  input: unknown;
  result: {
    successCount: number;
    errorCount: number;
    warningCount: number;
  };
}

export class HistoryStore {
  private buffer: HistoryEntry[] = [];
  private readonly capacity: number;

  constructor(capacity = 10) {
    this.capacity = capacity;
  }

  record(
    tool: ProtoToolName,
    input: unknown,
    result: HistoryEntry["result"],
  ): HistoryEntry | null {
    if (result.successCount === 0) return null;
    const entry: HistoryEntry = {
      historyId: randomUUID(),
      timestamp: Date.now(),
      tool,
      input,
      result,
    };
    this.buffer.push(entry);
    if (this.buffer.length > this.capacity) this.buffer.shift();
    return entry;
  }

  getLast(count = 1): HistoryEntry[] {
    if (count < 1) return [];
    return this.buffer.slice(-Math.min(count, this.buffer.length));
  }

  size(): number {
    return this.buffer.length;
  }
}

export function summarizeResult(raw: unknown): HistoryEntry["result"] {
  if (typeof raw !== "object" || raw === null) {
    return { successCount: 0, errorCount: 0, warningCount: 0 };
  }
  const r = raw as { successCount?: number; errorCount?: number; warningCount?: number };
  return {
    successCount: typeof r.successCount === "number" ? r.successCount : 0,
    errorCount: typeof r.errorCount === "number" ? r.errorCount : 0,
    warningCount: typeof r.warningCount === "number" ? r.warningCount : 0,
  };
}
```

#!/usr/bin/env node
import { Client } from "@modelcontextprotocol/sdk/client/index.js";
import { SSEClientTransport } from "@modelcontextprotocol/sdk/client/sse.js";

const URL_ = new URL("http://localhost:3000/sse");

async function withClient(fn) {
  const client = new Client({ name: "v1.21-probe", version: "0.0.0" }, { capabilities: {} });
  const transport = new SSEClientTransport(URL_);
  await client.connect(transport);
  try { return await fn(client); } finally { await client.close(); }
}

function unwrap(r) {
  const text = r?.content?.[0]?.text;
  if (text == null) return r;
  try { return JSON.parse(text); } catch { return text; }
}

async function call(client, name, args) {
  const r = await client.callTool({ name, arguments: args });
  if (r?.isError) { console.error(`[${name}] ERROR:`, r.content?.[0]?.text); process.exit(2); }
  return unwrap(r);
}

const [, , cmd, ...rest] = process.argv;

await withClient(async (client) => {
  if (cmd === "probe1") {
    const [from, to] = rest;
    await call(client, "proto_wire", { wires: [{ from, to }], replaceExisting: true });
    const h = await call(client, "proto_get_last_history", {});
    console.log("PROBE 1:", JSON.stringify(h, null, 2));
    return;
  }
  if (cmd === "probe2") {
    const [from1, overlayId, from2, to2] = rest;
    await call(client, "proto_overlay", { overlays: [{ mode: "open", from: from1, overlay: overlayId }], replaceExisting: true });
    await call(client, "proto_wire", { wires: [{ from: from2, to: to2 }], replaceExisting: true });
    const h = await call(client, "proto_get_last_history", { count: 2 });
    console.log("PROBE 2:", JSON.stringify(h, null, 2));
    return;
  }
  if (cmd === "probe3") {
    const [from, to] = rest;
    // 11 sequential proto_wire calls overwriting the same source-target pair (replaceExisting:true)
    for (let i = 0; i < 11; i++) {
      await call(client, "proto_wire", { wires: [{ from, to }], replaceExisting: true });
    }
    const h = await call(client, "proto_get_last_history", { count: 10 });
    console.log("PROBE 3 size:", h.entries.length);
    console.log("PROBE 3 timestamps:", h.entries.map(e => e.timestamp));
    return;
  }
  console.error("usage: node tmp/probe-v1.21.mjs probe1|probe2|probe3 <args>");
  process.exit(2);
});

### History stack
