# v1.7 Plugin Auto-Connect — Implementation Plan

> **For agentic workers:** Use superpowers:subagent-driven-development.

**Goal:** Persist the last-used channel via `figma.clientStorage` and auto-connect on plugin launch. Add a Disconnect button so the user can switch channels without restarting Figma.

**Architecture:** UI iframe ↔ plugin sandbox message protocol gains `load-channel` / `channel-loaded` / `save-channel` types. Sandbox owns the clientStorage round-trip. UI owns the auto-trigger + Disconnect button.

**Source spec:** `docs/superpowers/specs/2026-05-20-v1.7-plugin-auto-connect-design.md`

## Task 1: Sandbox handlers

**Files:**
- Modify: `src/figma-plugin/code.ts`

- [ ] **Step 1: Update `figma.ui.onmessage` handler**

The current handler reads:

```typescript
figma.ui.onmessage = (msg: any) => {
  if (msg?.type !== "command" || !msg.envelope) return;
  const envelope = msg.envelope as { id: string; command: Command["type"]; params: any };
  void commandQueue.enqueue(async () => {
    const response = await dispatch(envelope.command, envelope.params);
    figma.ui.postMessage({
      type: "response",
      envelope: { id: envelope.id, type: "response", ...response },
    });
  });
};
```

Replace with the multi-branch version:

```typescript
figma.ui.onmessage = (msg: any) => {
  if (msg?.type === "load-channel") {
    figma.clientStorage.getAsync("channel").then((value) => {
      figma.ui.postMessage({
        type: "channel-loaded",
        channel: typeof value === "string" && value.length > 0 ? value : null,
      });
    });
    return;
  }

  if (msg?.type === "save-channel" && typeof msg.channel === "string" && msg.channel.length > 0) {
    figma.clientStorage.setAsync("channel", msg.channel).catch(() => {
      // Storage failures are non-critical; user can re-type next session.
    });
    return;
  }

  if (msg?.type === "command" && msg.envelope) {
    const envelope = msg.envelope as { id: string; command: Command["type"]; params: any };
    void commandQueue.enqueue(async () => {
      const response = await dispatch(envelope.command, envelope.params);
      figma.ui.postMessage({
        type: "response",
        envelope: { id: envelope.id, type: "response", ...response },
      });
    });
  }
};
```

## Task 2: UI auto-connect + Disconnect

**Files:**
- Modify: `src/figma-plugin/ui.html`

- [ ] **Step 1: Replace the entire `<script>` block (and add a Disconnect button) in `ui.html`**

The current `ui.html` is small; the cleanest edit is to rewrite the body + script. Replace the section between `<body>` and `</body>` (inclusive of `<script>...</script>`) with:

```html
    <label for="channel">Channel</label>
    <input id="channel" placeholder="e.g. my-session" />
    <button id="connect">Connect</button>
    <button id="disconnect" style="display:none">Disconnect</button>
    <div id="status" class="pending">Loading…</div>

    <script>
      const channelInput = document.getElementById('channel');
      const connectBtn = document.getElementById('connect');
      const disconnectBtn = document.getElementById('disconnect');
      const statusEl = document.getElementById('status');
      let ws = null;

      function setStatus(text, cls) {
        statusEl.textContent = text;
        statusEl.className = cls;
      }

      function setConnectedUI(channel) {
        setStatus('Connected on channel: ' + channel, 'ok');
        channelInput.disabled = true;
        connectBtn.style.display = 'none';
        disconnectBtn.style.display = 'inline-block';
      }

      function setDisconnectedUI() {
        channelInput.disabled = false;
        connectBtn.style.display = 'inline-block';
        disconnectBtn.style.display = 'none';
      }

      function connect() {
        const channel = channelInput.value.trim();
        if (!channel) { setStatus('Channel name required', 'err'); return; }
        if (ws) ws.close();
        setStatus('Connecting…', 'pending');
        ws = new WebSocket('ws://localhost:3055');

        ws.onopen = () => {
          ws.send(JSON.stringify({ type: 'join', channel, id: 'plugin-join' }));
        };
        ws.onerror = () => setStatus('Connection failed', 'err');
        ws.onclose = () => {
          if (statusEl.className !== 'err') setStatus('Disconnected', 'err');
          setDisconnectedUI();
        };

        ws.onmessage = (e) => {
          let data;
          try { data = JSON.parse(e.data); } catch { return; }

          if (data.type === 'system' && data.message && data.message.id === 'plugin-join') {
            setConnectedUI(channel);
            parent.postMessage({ pluginMessage: { type: 'save-channel', channel } }, '*');
            return;
          }

          if (data.type === 'broadcast' && data.message && data.message.type === 'command') {
            parent.postMessage({ pluginMessage: { type: 'command', envelope: data.message } }, '*');
          }
        };
      }

      connectBtn.onclick = connect;

      disconnectBtn.onclick = () => {
        if (ws) ws.close();
        setStatus('Disconnected', 'err');
        setDisconnectedUI();
      };

      // Forward responses from the plugin sandbox out to WebSocket,
      // and react to channel-loaded events from the sandbox.
      window.onmessage = (event) => {
        const msg = event.data.pluginMessage;
        if (!msg) return;

        if (msg.type === 'channel-loaded') {
          if (typeof msg.channel === 'string' && msg.channel.length > 0) {
            channelInput.value = msg.channel;
            setStatus('Auto-connecting to ' + msg.channel + '…', 'pending');
            connect();
          } else {
            setStatus('Not connected', 'pending');
          }
          return;
        }

        if (msg.type === 'response' && ws && ws.readyState === WebSocket.OPEN) {
          ws.send(JSON.stringify({
            type: 'message',
            channel: channelInput.value.trim(),
            message: msg.envelope,
          }));
        }
      };

      // Request the last-used channel from sandbox on launch.
      parent.postMessage({ pluginMessage: { type: 'load-channel' } }, '*');
    </script>
```
