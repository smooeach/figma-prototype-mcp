- [ ] **Step 3: Implement the compile functions**

In `src/mcp-server/protoTools.ts`, append after the existing `compileProtoScroll` function (and before any final `export type` block if present):

```ts
export function compileProtoBack(input: ProtoBackInput): CreateReactionsInputType {
  const connections: Connection[] = input.backs.map((b) => {
    const action: Connection["action"] = { type: "back" };
    return buildConnection(b.from, b.trigger, b.motion, action);
  });
  return { connections, replaceExisting: input.replaceExisting };
}

export function compileProtoUrl(input: ProtoUrlInput): CreateReactionsInputType {
  const connections: Connection[] = input.urls.map((u) => {
    const action: Connection["action"] = {
      type: "url",
      url: u.url,
      openInNewTab: u.openInNewTab ?? false,
    };
    // Deliberately do NOT set `transition` — let create_reactions zod default it to "INSTANT".
    // The Connection type requires `transition` (post-parse), but we're returning the PRE-parse
    // shape that create_reactions consumes, where `transition: TransitionInput.default("INSTANT")`.
    return {
      sourceNodeId: u.from,
      trigger: u.trigger ?? DEFAULT_TRIGGER,
      action,
    } as Connection;
  });
  return { connections, replaceExisting: input.replaceExisting };
}
```

Note the type-cast on `compileProtoUrl`'s connection: TypeScript's inferred `CreateReactionsInputType` (from zod's `z.infer`) treats `transition` as required because post-parse it's never undefined (default applies). But we're emitting the input shape, where `transition` is optional. The `as Connection` cast acknowledges this PRE/POST parse asymmetry. The round-trip test (`CreateReactionsInput.safeParse(out).success`) confirms the cast is sound at runtime.

import { Client } from "@modelcontextprotocol/sdk/client/index.js";
import { SSEClientTransport } from "@modelcontextprotocol/sdk/client/sse.js";

const URL_ = new URL("http://localhost:3000/sse");

async function withClient(fn) {
  const client = new Client({ name: "v0.21-probe", version: "0.0.0" }, { capabilities: {} });
  const transport = new SSEClientTransport(URL_);
  await client.connect(transport);
  try { return await fn(client); } finally { await client.close(); }
}

function unwrap(r) {
  const text = r?.content?.[0]?.text;
  if (text == null) return r;
  try { return JSON.parse(text); } catch { return text; }
}

async function call(client, name, args) {
  const r = await client.callTool({ name, arguments: args });
  if (r?.isError) { console.error(`[${name}] ERROR:`, r.content?.[0]?.text); process.exit(2); }
  return unwrap(r);
}

const [, , cmd, ...rest] = process.argv;

await withClient(async (client) => {
  if (cmd === "list") {
    const list = await client.listTools();
    console.log("Tools:", list.tools.map((t) => t.name).join(", "));
    console.log("Count:", list.tools.length);
    return;
  }
  if (cmd === "back") {

    if (!from) { console.error("usage: back <fromNodeId>"); process.exit(2); }
    console.log(`[probe 1] proto_back ${from}, then proto_get_last_history`);
