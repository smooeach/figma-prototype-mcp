**Goal:** Extend v1.17's `set_variable` to accept hex string values (`#RRGGBB` / `#RRGGBBAA`) for COLOR-type Figma variables. Variable name resolution, error messages, and `list_reactions` round-trip via hex.

**Tech Stack:** TypeScript + Figma Plugin API (`Variable.resolvedType === "COLOR"`, `figma.variables.getLocalVariablesAsync`).

```typescript
const HEX_REGEX = /^#[0-9a-fA-F]{6}([0-9a-fA-F]{2})?$/;

function hexToRgb(hex: string): { r: number; g: number; b: number; a?: number } {
  // Caller must validate against HEX_REGEX before calling.
  const clean = hex.slice(1);
  const r = parseInt(clean.slice(0, 2), 16) / 255;
  const g = parseInt(clean.slice(2, 4), 16) / 255;
  const b = parseInt(clean.slice(4, 6), 16) / 255;
  if (clean.length === 8) {
    const a = parseInt(clean.slice(6, 8), 16) / 255;
    return { r, g, b, a };
  }
  return { r, g, b };
}

function rgbToHex(rgb: { r: number; g: number; b: number; a?: number }): string {
  const toHex = (n: number) =>
    Math.max(0, Math.min(255, Math.round(n * 255))).toString(16).padStart(2, "0").toUpperCase();
  const base = `#${toHex(rgb.r)}${toHex(rgb.g)}${toHex(rgb.b)}`;
  if (rgb.a === undefined || Math.round(rgb.a * 255) === 255) return base;
  return base + toHex(rgb.a);
}
```

```typescript
/**
 * Validate that a JS literal's type matches a Figma Variable's resolvedType
 * and return the corresponding VariableData wrapping.
 * Throws with a precise message on mismatch or unsupported variable type.
 * Shared by conditional's buildCondition and set_variable's buildSetVariableData.
 *
 * COLOR variables accept hex string values (#RRGGBB or #RRGGBBAA) for
 * assignment context. Conditional comparison against COLOR variables is
 * rejected (v1.18 non-goal).
 */
function validateVariableLiteralCompat(
  variable: Variable,
  value: boolean | number | string,
  context: string,                       // "comparison" or "assignment"
): unknown {
  const valueType = typeof value;

  if (variable.resolvedType === "COLOR") {
    if (context === "comparison") {
      throw new Error(`Variable "${variable.name}" is COLOR; conditional comparison against COLOR variables is not supported in v1.18 (use BOOLEAN/FLOAT/STRING for condition.value)`);
    }
    // assignment context
    if (valueType !== "string") {
      throw new Error(`Variable "${variable.name}" is COLOR; cannot assign ${valueType} literal (expected hex string)`);
    }
    if (!HEX_REGEX.test(value as string)) {
      throw new Error(`Variable "${variable.name}" is COLOR; value must be a hex string like #RRGGBB or #RRGGBBAA (got "${value}")`);
    }
    return { type: "COLOR", resolvedType: "COLOR", value: hexToRgb(value as string) };
  }

  const expected =
    variable.resolvedType === "BOOLEAN" ? "boolean"
    : variable.resolvedType === "FLOAT" ? "number"
    : variable.resolvedType === "STRING" ? "string"
    : "unknown";
  if (expected === "unknown") {
    throw new Error(`Variable "${variable.name}" has unsupported type ${variable.resolvedType} for ${context} (supported: BOOLEAN, FLOAT, STRING, COLOR)`);
  }
  if (valueType !== expected) {
    const action = context === "comparison" ? "compare against" : "assign";
    throw new Error(`Variable "${variable.name}" is ${variable.resolvedType}; cannot ${action} ${valueType} literal (expected ${expected})`);
  }
  return {
    type: variable.resolvedType,
    resolvedType: variable.resolvedType,
    value,
  };
}
```

Key changes from previous:
- Return type narrowed to `unknown` (was `{ type: BOOLEAN|FLOAT|STRING; ... }`).
- New COLOR branch handles both `context === "comparison"` (reject) and `context === "assignment"` (hex parse + return COLOR VariableData).
- Updated "unsupported type" error message to mention COLOR among the supported list.

const ws = new WebSocket(RELAY_URL);
const pending = new Map();
function send(command, params) {
  const id = randomUUID();
  return new Promise((resolve, reject) => {
    const t = setTimeout(() => { pending.delete(id); reject(new Error(`${command} timeout`)); }, 15_000);
    pending.set(id, { resolve, reject, timer: t });
    ws.send(JSON.stringify({ type: "message", channel: CHANNEL, message: { id, type: "command", command, params } }));
  });
}
ws.on("message", (raw) => {
  let d; try { d = JSON.parse(raw.toString()); } catch { return; }
  if (d.type === "system" && d.message?.id === "mcp-join") { console.log(`[join] ${d.message.result ?? "ok"}`); return; }
  if (d.type !== "broadcast") return;
  const r = d.message; if (!r || r.type !== "response") return;
  const p = pending.get(r.id); if (!p) return;
  pending.delete(r.id); clearTimeout(p.timer);
  if (r.status === "ok") p.resolve(r.result); else p.reject(new Error(r.error?.message ?? "plugin error"));
});
