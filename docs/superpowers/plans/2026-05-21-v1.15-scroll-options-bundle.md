> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** `set_frame_scroll`에 optional `fixedChildren?: number` (sticky header) 추가 + 4 NODE action (navigate/scroll/overlay/swap_overlay)에 optional `resetScrollPosition?: boolean` 추가. 동시에 builder가 **현재 hardcode 중인 deprecated `preserveScrollPosition: false`** 를 제거.

**Tech Stack:** TypeScript + Zod + Vitest + Figma Plugin API (`FrameNode.numberOfFixedChildren`, NODE Action's `resetScrollPosition`).

**Source spec:** `docs/superpowers/specs/2026-05-21-v1.15-scroll-options-bundle-design.md`

기존 `handleSetFrameScroll` 함수 전체를 다음으로 교체:

```typescript
async function handleSetFrameScroll(params: {
  frames: Array<{
    frameId: string;
    direction?: "NONE" | "HORIZONTAL" | "VERTICAL" | "BOTH";
    fixedChildren?: number;
  }>;
}) {
  await figma.loadAllPagesAsync();
  const results: Array<{
    frameId: string;
    status: "success" | "error";
    applied?: string[];
    error?: string;
  }> = [];
  let successCount = 0;
  let errorCount = 0;
  for (const { frameId, direction, fixedChildren } of params.frames) {
    try {
      const node = figma.getNodeById(frameId);
      if (!node) throw new Error(`Frame not found: ${frameId}`);
      if (node.type !== "FRAME") {
        throw new Error(`Node is not a FRAME: ${node.name} (type: ${node.type})`);
      }
      const applied: string[] = [];
      if (direction !== undefined) {
        (node as FrameNode).overflowDirection = direction;
        applied.push("direction");
      }
      if (fixedChildren !== undefined) {
        (node as FrameNode).numberOfFixedChildren = fixedChildren;
        applied.push("fixedChildren");
      }
      results.push({ frameId, status: "success", applied });
      successCount++;
    } catch (e: any) {
      results.push({ frameId, status: "error", error: e?.message ?? String(e) });
      errorCount++;
    }
  }
  return { results, successCount, errorCount };
}
```

```javascript
import WebSocket from "ws";
import { randomUUID } from "node:crypto";

const RELAY_URL = process.env.FIGMA_RELAY_URL ?? "ws://localhost:3055";
const CHANNEL   = process.env.FIGMA_CHANNEL   ?? "test1";

const VTALL = "995:3"; // from scenario 19 fixture

const cases = [
  { name: "20a  VERTICAL + fixedChildren:1", frames: [{ frameId: VTALL, direction: "VERTICAL", fixedChildren: 1 }] },
  { name: "20b  fixedChildren:0 only",       frames: [{ frameId: VTALL, fixedChildren: 0 }] },
  { name: "20d  fixedChildren:999 (OOR)",    frames: [{ frameId: VTALL, fixedChildren: 999 }] },
  // 20c is Zod-level — verified by unit test, no live wire needed
  // 20e is Zod refine — verified by unit test
];

const ws = new WebSocket(RELAY_URL);
const pending = new Map();

function send(command, params) {
  const id = randomUUID();
  return new Promise((resolve, reject) => {
    const timer = setTimeout(() => { pending.delete(id); reject(new Error(`${command} timeout`)); }, 15_000);
    pending.set(id, { resolve, reject, timer });
    ws.send(JSON.stringify({ type: "message", channel: CHANNEL, message: { id, type: "command", command, params } }));
  });
}

ws.on("message", (raw) => {
  let data; try { data = JSON.parse(raw.toString()); } catch { return; }
  if (data.type === "system" && data.message?.id === "mcp-join") { console.log(`[join] ${data.message.result ?? "ok"}`); return; }
  if (data.type !== "broadcast") return;
  const resp = data.message;
  if (!resp || resp.type !== "response") return;
  const p = pending.get(resp.id);
  if (!p) return;
  pending.delete(resp.id); clearTimeout(p.timer);
  if (resp.status === "ok") p.resolve(resp.result); else p.reject(new Error(resp.error?.message ?? "plugin error"));
});

ws.on("open", async () => {
  ws.send(JSON.stringify({ type: "join", channel: CHANNEL, id: "mcp-join" }));
  await new Promise((r) => setTimeout(r, 250));
  for (const c of cases) {
    process.stdout.write(`--- ${c.name} ---\n`);
    try {
      const r = await send("SET_FRAME_SCROLL", { frames: c.frames });
      console.log(JSON.stringify(r, null, 2));
    } catch (e) {
      console.log("CMD ERROR:", e.message);
    }
  }
  ws.close();
});
ws.on("close", () => process.exit(0));
ws.on("error", (e) => { console.error("ws error:", e.message); process.exit(1); });
```
